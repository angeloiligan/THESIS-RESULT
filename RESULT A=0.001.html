def convert_rsi(val):
    if val == "R":
        return 1
    elif val == "S":
        return -1
    elif val == "I":
        return 1

Ising Energy Function

In [1258]:
def ising_energy(S, J, h):
    interaction = 0.5 * np.sum(J * np.outer(S, S))
    field = np.dot(h, S)
    return interaction + field

Estimate Temperature

In [1264]:
def kl_divergence(p, q):
    eps = 1e-10
    return np.sum(p * np.log((p + eps) / (q + eps)))

T_values = np.linspace(0.1, 5, 100)

best_T = None
best_score = np.inf

for T_test in T_values:
    boltz = np.exp(-config_df['Energy'] / T_test)
    P_model = boltz / boltz.sum()

    score = kl_divergence(config_df['P_data'].values, P_model)

    if score < best_score:
        best_score = score
        best_T = T_test

print("KL Divergence:", best_score)

from numba import njit
@njit
def monte_carlo_step(spins, J, h, T):
    n = len(spins)

    for _ in range(n):
        i = np.random.randint(0, n)

        dE = 0
        for j in range(n):
            if i != j:
                dE += 2 * J[i, j] * spins[i] * spins[j]
        dE += 2 * h[i] * spins[i]

        # Metropolis acceptance
        if dE <= 0 or np.random.rand() < np.exp(-dE / T):
            spins[i] *= -1

J = J_matrix.values
h = h_vector.values
N = len(h)

T_values = np.linspace(0.5, 5, 25)

M_vals = []
E_vals = []
chi_vals = []
C_vals = []

for T in T_values:
    spins = np.random.choice([-1, 1], size=N)

    for _ in range(2000):
        monte_carlo_step(spins, J, h, T)

    energies = []
    mags = []

    for _ in range(10000):
        monte_carlo_step(spins, J, h, T)

        E = 0.5 * np.sum(J * np.outer(spins, spins)) + np.dot(h, spins)
        M = np.sum(spins) / N

        energies.append(E)
        mags.append(M)

    energies = np.array(energies)
    mags = np.array(mags)

    E_mean = np.mean(energies)
    M_mean = np.mean(np.abs(mags))

    chi = (np.mean(mags**2) - np.mean(mags)**2) * N / T
    C = (np.mean(energies**2) - E_mean**2) / (T**2)

    E_vals.append(E_mean / N)
    M_vals.append(M_mean)
    chi_vals.append(chi)
    C_vals.append(C)
